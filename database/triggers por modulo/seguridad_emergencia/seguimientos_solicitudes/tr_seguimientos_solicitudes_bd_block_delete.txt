Explica: trigger BEFORE DELETE sobre la tabla seguimientos_solicitudes.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_seguimientos_solicitudes_bd_block_delete//
CREATE TRIGGER tr_seguimientos_solicitudes_bd_block_delete BEFORE DELETE ON seguimientos_solicitudes
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla seguimientos_solicitudes. Use eliminacion logica.';
END
//
DELIMITER ;
