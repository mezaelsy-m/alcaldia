Explica: trigger BEFORE DELETE sobre la tabla seguridad.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_seguridad_bd_block_delete//
CREATE TRIGGER tr_seguridad_bd_block_delete BEFORE DELETE ON seguridad
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla seguridad. Use eliminacion logica.';
END
//
DELIMITER ;
