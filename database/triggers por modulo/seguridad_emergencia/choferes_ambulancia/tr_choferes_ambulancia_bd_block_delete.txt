Explica: trigger BEFORE DELETE sobre la tabla choferes_ambulancia.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_choferes_ambulancia_bd_block_delete//
CREATE TRIGGER tr_choferes_ambulancia_bd_block_delete BEFORE DELETE ON choferes_ambulancia
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla choferes_ambulancia. Use eliminacion logica.';
END
//
DELIMITER ;
