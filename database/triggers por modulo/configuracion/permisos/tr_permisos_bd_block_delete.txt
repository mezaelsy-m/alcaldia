Explica: trigger BEFORE DELETE sobre la tabla permisos.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_permisos_bd_block_delete//
CREATE TRIGGER tr_permisos_bd_block_delete BEFORE DELETE ON permisos
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla permisos. Use eliminacion logica.';
END
//
DELIMITER ;
