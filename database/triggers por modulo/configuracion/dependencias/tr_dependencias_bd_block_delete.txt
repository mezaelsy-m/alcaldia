Explica: trigger BEFORE DELETE sobre la tabla dependencias.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_dependencias_bd_block_delete//
CREATE TRIGGER tr_dependencias_bd_block_delete BEFORE DELETE ON dependencias
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla dependencias. Use eliminacion logica.';
END
//
DELIMITER ;
