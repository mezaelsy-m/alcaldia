Explica: trigger BEFORE DELETE sobre la tabla tipos_ayuda_social.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_tipos_ayuda_social_bd_block_delete//
CREATE TRIGGER tr_tipos_ayuda_social_bd_block_delete BEFORE DELETE ON tipos_ayuda_social
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla tipos_ayuda_social. Use eliminacion logica.';
END
//
DELIMITER ;
