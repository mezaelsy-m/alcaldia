Explica: trigger BEFORE DELETE sobre la tabla usuarios.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_usuarios_bd_block_delete//
CREATE TRIGGER tr_usuarios_bd_block_delete BEFORE DELETE ON usuarios
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla usuarios. Use eliminacion logica.';
END
//
DELIMITER ;
