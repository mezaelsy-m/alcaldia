Explica: trigger BEFORE DELETE sobre la tabla empleados.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_empleados_bd_block_delete//
CREATE TRIGGER tr_empleados_bd_block_delete BEFORE DELETE ON empleados
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla empleados. Use eliminacion logica.';
END
//
DELIMITER ;
