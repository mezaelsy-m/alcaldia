Explica: consulta un registro puntual por clave primaria y permite listar registros para gestion.
SQL (detalle por id):
SELECT id_configuracion_smtp,
       host,
       puerto,
       usuario,
       clave,
       correo_remitente,
       nombre_remitente,
       usar_tls,
       estado,
       id_usuario_actualiza,
       fecha_registro,
       fecha_actualizacion
FROM configuracion_smtp
WHERE id_configuracion_smtp = ?;

SQL (listado):
SELECT id_configuracion_smtp,
       host,
       puerto,
       usuario,
       clave,
       correo_remitente,
       nombre_remitente,
       usar_tls,
       estado,
       id_usuario_actualiza,
       fecha_registro,
       fecha_actualizacion
FROM configuracion_smtp
WHERE estado = 1
ORDER BY id_configuracion_smtp DESC;
