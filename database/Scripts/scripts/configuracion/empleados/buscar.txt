Explica: consulta un registro puntual por clave primaria y permite listar registros para gestion.
SQL (detalle por id):
SELECT id_empleado,
       cedula,
       nombre,
       apellido,
       id_dependencia,
       telefono,
       correo,
       direccion,
       estado
FROM empleados
WHERE id_empleado = ?;

SQL (listado):
SELECT id_empleado,
       cedula,
       nombre,
       apellido,
       id_dependencia,
       telefono,
       correo,
       direccion,
       estado
FROM empleados
WHERE estado = 1
ORDER BY id_empleado DESC;
