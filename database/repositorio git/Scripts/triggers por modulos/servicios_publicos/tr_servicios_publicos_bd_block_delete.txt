Explica: trigger BEFORE DELETE sobre la tabla servicios_publicos.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_servicios_publicos_bd_block_delete//
CREATE TRIGGER tr_servicios_publicos_bd_block_delete BEFORE DELETE ON servicios_publicos
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla servicios_publicos. Use eliminacion logica.';
END
//
DELIMITER ;
