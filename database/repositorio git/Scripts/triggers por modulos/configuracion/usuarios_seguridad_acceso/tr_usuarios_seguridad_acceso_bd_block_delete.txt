Explica: trigger BEFORE DELETE sobre la tabla usuarios_seguridad_acceso.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_usuarios_seguridad_acceso_bd_block_delete//
CREATE TRIGGER tr_usuarios_seguridad_acceso_bd_block_delete BEFORE DELETE ON usuarios_seguridad_acceso
FOR EACH ROW
BEGIN

DELIMITER ;
