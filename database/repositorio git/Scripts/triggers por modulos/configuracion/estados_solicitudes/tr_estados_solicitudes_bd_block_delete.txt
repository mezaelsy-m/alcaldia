Explica: trigger BEFORE DELETE sobre la tabla estados_solicitudes.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_estados_solicitudes_bd_block_delete//
CREATE TRIGGER tr_estados_solicitudes_bd_block_delete BEFORE DELETE ON estados_solicitudes
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla estados_solicitudes. Use eliminacion logica.';
END
//
DELIMITER ;
