Explica: trigger AFTER UPDATE sobre la tabla beneficiarios.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_beneficiarios_au_audit//
CREATE TRIGGER tr_beneficiarios_au_audit AFTER UPDATE ON beneficiarios
FOR EACH ROW
BEGIN
  INSERT INTO bitacora (id_usuario, tabla_afectada, accion, id_registro, resumen, detalle, datos_antes, datos_despues, usuario_bd, fecha_evento, estado)
  VALUES (NULL, 'beneficiarios', 'UPDATE', CAST(NEW.id_beneficiario AS CHAR), 'UPDATE en beneficiarios', 'Se actualizo un registro en beneficiarios', JSON_OBJECT('id_beneficiario', OLD.id_beneficiario, 'nacionalidad', OLD.nacionalidad, 'cedula', OLD.cedula, 'nombre_beneficiario', OLD.nombre_beneficiario, 'telefono', OLD.telefono, 'id_comunidad', OLD.id_comunidad, 'comunidad', OLD.comunidad, 'fecha_registro', OLD.fecha_registro, 'estado', OLD.estado), JSON_OBJECT('id_beneficiario', NEW.id_beneficiario, 'nacionalidad', NEW.nacionalidad, 'cedula', NEW.cedula, 'nombre_beneficiario', NEW.nombre_beneficiario, 'telefono', NEW.telefono, 'id_comunidad', NEW.id_comunidad, 'comunidad', NEW.comunidad, 'fecha_registro', NEW.fecha_registro, 'estado', NEW.estado), CURRENT_USER(), NOW(), 1);
END
//
DELIMITER ;
