    ruta_archivo = ?,
    estado_envio = ?,
    correo_destino = ?,
    fecha_envio = ?,
    detalle_envio = ?,
    id_usuario_genera = ?,
    fecha_generacion = ?,
    estado = ?
WHERE id_reporte_solicitud = ?;
