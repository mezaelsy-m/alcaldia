Explica: trigger BEFORE DELETE sobre la tabla reportes_solicitudes_ambulancia.
SQL:
DELIMITER //
DROP TRIGGER IF EXISTS tr_reportes_solicitudes_ambulancia_bd_block_delete//
CREATE TRIGGER tr_reportes_solicitudes_ambulancia_bd_block_delete BEFORE DELETE ON reportes_solicitudes_ambulancia
FOR EACH ROW
BEGIN
  SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eliminacion fisica bloqueada en tabla reportes_solicitudes_ambulancia. Use eliminacion logica.';
END
//
DELIMITER ;
