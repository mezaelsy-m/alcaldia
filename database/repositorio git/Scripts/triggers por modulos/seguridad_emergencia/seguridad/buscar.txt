SELECT id_seguridad,
       ticket_interno,
       id_beneficiario,
       beneficiario,
       id_usuario,
       id_tipo_seguridad,
       tipo_seguridad,
       requiere_ambulancia,
       id_solicitud_seguridad,
       tipo_solicitud,
       id_estado_solicitud,
       estado_solicitud,
       codigo_estado_solicitud,
       clase_badge_estado_solicitud,
       fecha_seguridad,
       fecha_seguridad_formateada,
       descripcion,
       estado_atencion,
       ubicacion_evento,
       referencia_evento,
       id_despacho_unidad,

       modo_asignacion,

       referencia_actual,
       id_chofer_ambulancia,
