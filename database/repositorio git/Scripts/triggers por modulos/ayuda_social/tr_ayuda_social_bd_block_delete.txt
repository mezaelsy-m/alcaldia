Explica: trigger BEFORE DELETE sobre la tabla ayuda_social.

DELIMITER //
DROP TRIGGER IF EXISTS tr_ayuda_social_bd_block_delete//
CREATE TRIGGER tr_ayuda_social_bd_block_delete BEFORE DELETE ON ayuda_social
FOR EACH ROW
BEGIN

END
